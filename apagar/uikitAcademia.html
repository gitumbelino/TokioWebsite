A academia que conhece o caminho — porque já o percorreu — e dá aos alunos o caminho mais directo e comprovado para passar no exame.

05 — Código
variables.css
/* ============================================
   Academia de Estudos — Design Tokens
   variables.css
   ============================================ */

:root {

  /* --- TYPOGRAPHY --- */
  --font-heading: 'Barlow Condensed', sans-serif;
  --font-body:    'Barlow', sans-serif;

  /* --- NAVY SCALE --- */
  --color-navy-deep:   #0C1B2E; /* hero, footer, navbar */
  --color-navy-mid:    #1A3354; /* cards, section dividers */
  --color-navy-light:  #3B6080; /* overlays, testimonials */
  --color-navy-muted:  #8BAFC8; /* secondary text on dark */

  /* --- NEUTRAL / SURFACES --- */
  --color-surface-base:   #FFFFFF; /* main page background */
  --color-surface-subtle: #F6F8FA; /* alternating sections */
  --color-surface-card:   #EAEFF4; /* cards, table rows, inputs */
  --color-text-base:      #1E1E1E; /* body text on light */

  /* --- ACCENT HIERARCHY --- */
  --color-accent-primary:  #E05B3A; /* ember — CTAs, highlights */
  --color-accent-prestige: #C8A84B; /* gold — badges, stats */
  --color-accent-utility:  #00897B; /* teal — links, icons */

  /* --- SPACING (8pt grid) --- */
  --space-1:  8px;
  --space-2:  16px;
  --space-3:  24px;
  --space-4:  32px;
  --space-6:  48px;
  --space-8:  64px;
  --space-12: 96px;

  /* --- LAYOUT --- */
  --max-width:    1100px;
  --radius-sm:    4px;
  --radius-md:    8px;
  --radius-lg:    12px;
  --border-color: #EAEFF4;

  /* --- TYPE SCALE --- */
  --text-h1:      clamp(36px, 6vw, 60px);
  --text-h2:      clamp(26px, 4vw, 36px);
  --text-h3:      clamp(20px, 3vw, 26px);
  --text-h4:      20px;
  --text-body:    16px;
  --text-small:   14px;
  --text-caption: 13px;
}
